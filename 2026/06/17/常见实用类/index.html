一、字符串相关：String、StringBuilder、StringBuffer

1. String

// 特点：字符串不可变，底层byte数组，存在常量池复用
String s1 = "abc";
String s2 = new String("abc");

// 常用方法
s1.length();          // 获取字符串长度
s1.equals("abc");     // 比较内容是否相等
s1.contains("ab");    // 判断是否包含子串
s1.substring(0,2);   // 截取字符串
s1.split(",");        // 按符号分割成数组
s1.trim();            // 去除首尾空格
s1.toUpperCase();     // 转大写
s1.toLowerCase();     // 转小写

2. StringBuilder
java
运行
// 线程不安全、拼接效率高，大量字符串拼接首选
StringBuilder sb = new StringBuilder();
sb.append("a");       // 拼接内容
sb.reverse();         // 字符串反转
sb.toString();        // 转为String

3. StringBuffer
java
运行
// 方法与StringBuilder完全一致，加synchronized同步锁，线程安全，性能更低
StringBuffer sb = new StringBuffer();

二、数学工具类：Math
java
运行
// 全部静态方法，直接类名调用
Math.abs(-10);        // 取绝对值
Math.max(3,9);        // 获取最大值
Math.min(3,9);        // 获取最小值
Math.random();        // [0,1)随机小数
Math.round(3.6);      // 四舍五入
Math.ceil(3.1);       // 向上取整
Math.floor(3.9);      // 向下取整
Math.PI;              // 圆周率常量

三、控制台输入：Scanner
java
运行
import java.util.Scanner;
Scanner sc = new Scanner(System.in);

sc.nextInt();     // 读取整数
sc.nextDouble();  // 读取小数
sc.next();        // 读取一段字符串（空格截止）
sc.nextLine();    // 读取整行字符串
sc.close();       // 关闭输入流

四、数组工具类：Arrays
java
运行
import java.util.Arrays;
int[] arr = {5,1,3};

Arrays.toString(arr);     // 数组转字符串打印
Arrays.sort(arr);         // 数组升序排序
Arrays.copyOf(arr, 5);    // 数组拷贝
Arrays.equals(a1,a2);     // 判断两个数组内容是否一致

五、时间日期类（JDK8 推荐）
1. LocalDate 日期（年月日）
java
运行
LocalDate today = LocalDate.now();
LocalDate date = LocalDate.of(2026,6,17);
today.getYear();
today.getMonthValue();
today.getDayOfMonth();
today.plusDays(3);  // 加3天

2. LocalTime 时间（时分秒）
java
运行
LocalTime now = LocalTime.now();
LocalTime time = LocalTime.of(14,30);

3. LocalDateTime 日期 + 时间
java
运行
LocalDateTime ldt = LocalDateTime.now();
// 格式化输出
DateTimeFormatter fmt = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss");
ldt.format(fmt);

六、空值处理：Optional
java
运行
// 避免空指针异常
String str = null;
Optional<String> opt = Optional.ofNullable(str);
opt.orElse("默认值");    // 为空返回默认值
opt.isPresent();         // 判断是否不为null

七、系统工具类：System
java
运行
System.out.println();  // 控制台输出
System.currentTimeMillis(); // 获取当前毫秒时间戳
System.exit(0);        // 终止程序
System.arraycopy();    // 数组底层拷贝

八、包装类（基本类型与字符串互转）
java
运行
// 8种包装类：Byte Short Integer Long Float Double Character Boolean
Integer num = Integer.valueOf("123"); // 字符串转数字
String s = Integer.toString(123);     // 数字转字符串

// 自动装箱、拆箱
Integer a = 10;  // 装箱
int b = a;        // 拆箱
